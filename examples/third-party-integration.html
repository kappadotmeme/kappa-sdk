Kappa Deployer Widget - Integration Examples

1. Basic HTML Integration (CDN)

The simplest way to integrate the Deployer Widget in a plain HTML page:

<!-- Include React and the Widget -->
<script crossorigin src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
<script crossorigin src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
<script src="https://unpkg.com/kappa-sdk@latest/dist/deployer-widget.min.js"></script>

<!-- Container for the widget -->
<div id="deployer-widget"></div>

<script>
const widget = KappaDeployer.DeployerWidgetStandalone({
network: {
bondingContract: '0xf1ba7eae2494f147cf4a67e8f87b894382ebe9261c5f1cd7c13fdacce82ebc37',
CONFIG: '0x9b0fb19055c8b77f76203635ef6c4b4dac9928031d42c7e42131491adc3f87ae'
},
onSuccess: function(coinAddress) {
console.log('Token created:', coinAddress);
window.location.href = 'https://kappa.fun/coin/' + coinAddress;
},
defaultDevBuySui: '0.1',
projectName: 'My Project'
});

ReactDOM.render(widget, document.getElementById('deployer-widget'));
</script>

2. React App Integration (NPM)

For React applications using npm/yarn:

// Install the package
npm install kappa-sdk

// In your React component
import { DeployerWidgetStandalone } from 'kappa-sdk/react';

function App() {
return (
<DeployerWidgetStandalone
network={{
bondingContract: '0xf1ba7eae2494f147cf4a67e8f87b894382ebe9261c5f1cd7c13fdacce82ebc37',
CONFIG: '0x9b0fb19055c8b77f76203635ef6c4b4dac9928031d42c7e42131491adc3f87ae'
}}
onSuccess={(coinAddress) => {
console.log('Token created:', coinAddress);
}}
defaultDevBuySui="0.1"
logoUrl="/logo.png"
projectName="My DeFi Project"
/>
);
}

3. Next.js Integration

For Next.js applications with SSR:

// Use dynamic import to avoid SSR issues
import dynamic from 'next/dynamic';

const DeployerWidget = dynamic(
() => import('kappa-sdk/react').then(mod => mod.DeployerWidgetStandalone),
{ ssr: false }
);

export default function TokenLauncher() {
return (
<DeployerWidget
network={{
bondingContract: '0xf1ba7eae2494f147cf4a67e8f87b894382ebe9261c5f1cd7c13fdacce82ebc37',
CONFIG: '0x9b0fb19055c8b77f76203635ef6c4b4dac9928031d42c7e42131491adc3f87ae'
}}
onSuccess={(coinAddress) => {
router.push(`/token/${coinAddress}`);
}}
/>
);
}

4. Custom Themed Integration

Customize the widget appearance to match your brand:

<DeployerWidgetStandalone
network={{
bondingContract: '0xf1ba7eae2494f147cf4a67e8f87b894382ebe9261c5f1cd7c13fdacce82ebc37',
CONFIG: '0x9b0fb19055c8b77f76203635ef6c4b4dac9928031d42c7e42131491adc3f87ae'
}}
theme={{
'--kappa-bg': '#1a1a2e',
'--kappa-panel': '#16213e',
'--kappa-border': '#0f3460',
'--kappa-text': '#ffffff',
'--kappa-primary': '#e94560',
'--kappa-success': '#00d4aa',
'--kappa-accent': '#533483'
}}
logoUrl="https://yoursite.com/logo.png"
projectName="Your Brand"
defaultDevBuySui="0.5"
maxWidth={600}
/>

Important Notes:

  • The widget requires a Web3 wallet (Sui Wallet, Suiet, etc.) to be installed
  • Make sure to use the correct network configuration for mainnet or testnet
  • The onSuccess callback receives the full coin address that can be used to redirect users
  • All styling is contained within the widget and won't affect your app's styles
  • The widget is fully responsive and works on mobile devices

5. TypeScript Integration

Full TypeScript support with type definitions:

import {
DeployerWidgetStandalone,
DeployerWidgetProps
} from 'kappa-sdk/react';

const widgetConfig: DeployerWidgetProps = {
network: {
bondingContract: '0xf1ba7eae2494f147cf4a67e8f87b894382ebe9261c5f1cd7c13fdacce82ebc37',
CONFIG: '0x9b0fb19055c8b77f76203635ef6c4b4dac9928031d42c7e42131491adc3f87ae'
},
onSuccess: (coinAddress: string): void => {
console.log('Created token:', coinAddress);
},
defaultDevBuySui: '0.1',
maxWidth: 500,
theme: {
'--kappa-primary': '#2563eb'
}
};

function App(): JSX.Element {
return <DeployerWidgetStandalone {...widgetConfig} />;
}
